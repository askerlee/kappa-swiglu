options:
  -h, --help            show this help message and exit
  --device-type DEVICE_TYPE
                        cuda|cpu|mps (empty = autodetect)
  --seed SEED           random seed for initialization
  --mockup-mode [MOCKUP_MODE]
                        skip actual training/eval/sample compute and only
                        advance step counter
  --fp8 [FP8]           enable FP8 training (requires H100+ GPU and torchao)
  --fp8-recipe {rowwise,tensorwise}
                        FP8 scaling recipe: tensorwise (faster, recommended)
                        or rowwise (more accurate but slower)
  --depth DEPTH         depth of the Transformer model
  --moe-start-layer MOE_START_LAYER
                        first layer index of MoE layers
  --num-moe-layers NUM_MOE_LAYERS
                        number of MoE layers to instantiate from --moe-start-
                        layer onward (-1 = all eligible layers)
  --n-exp N_EXP         number of experts per MoE layer
  --moe-top-k MOE_TOP_K
                        top-k of the MoE routing
  --use-aux-free-load-balancing [USE_AUX_FREE_LOAD_BALANCING]
                        enable DeepSeekV3 auxiliary-loss-free load balancing
                        instead of the Switch auxiliary router loss
  --aux-loss-weight AUX_LOSS_WEIGHT
                        final weight for the Switch-style router auxiliary
                        load-balancing loss after the initial 500-step anneal
  --aux-loss-weight-init-scale AUX_LOSS_WEIGHT_INIT_SCALE
                        initial aux loss weight scale factor; the anneal
                        starts from --aux-loss-weight * this value
  --aux-loss-weight-init-anneal-iterations AUX_LOSS_WEIGHT_INIT_ANNEAL_ITERATIONS
                        number of iterations used to anneal aux loss weight
                        from --aux-loss-weight * --aux-loss-weight-init-scale
                        down to --aux-loss-weight
  --use-exp-gate-proj-bias [USE_EXP_GATE_PROJ_BIAS]
                        add a learnable bias to Qwen3 expert gate activations
                        after gate_proj and SiLU
  --exp-gate-proj-bias-input {top_logits,router_probs}
                        router confidence signal used by expert
                        gate_proj_bias: raw selected logits or top-k router
                        probabilities
  --gate-proj-bias-start-layer GATE_PROJ_BIAS_START_LAYER
                        first transformer layer index where MoE gate_proj_bias
                        is enabled (default: when omitted and MoE is enabled,
                        use min(moe_start_layer + 2, depth//2, 5))
  --gate-proj-bias-lr-max-scale GATE_PROJ_BIAS_LR_MAX_SCALE
                        peak LR scale factor for gate_proj_bias params after
                        warming from 0 before annealing to --gate-proj-bias-
                        lr-final-scale
  --gate-proj-bias-lr-final-scale GATE_PROJ_BIAS_LR_FINAL_SCALE
                        final LR scale factor for gate_proj_bias params after
                        warming from 0 to 1
  --gate-proj-bias-delay-start-iterations GATE_PROJ_BIAS_DELAY_START_ITERATIONS
                        number of initial iterations to keep gate_proj_bias LR
                        at 0 before warmup and annealing
  --gate-proj-bias-lr-warmup-iterations GATE_PROJ_BIAS_LR_WARMUP_ITERATIONS
                        number of iterations to linearly ramp gate_proj_bias
                        LR scale from 0 to --gate-proj-bias-lr-max-scale
                        before annealing to --gate-proj-bias-lr-final-scale
  --gate-proj-bias-l2-loss-weight GATE_PROJ_BIAS_L2_LOSS_WEIGHT
                        weight for MoE gate_proj_bias L2 loss
  --gate-proj-bias-l2-loss-anneal-iterations GATE_PROJ_BIAS_L2_LOSS_ANNEAL_ITERATIONS
                        iterations for stage-1 anneal of the MoE (2D)
                        gate_proj_bias L2 loss from 1.0 to --gate-proj-
                        bias-l2-loss-stage1-frac (-1 = use half total training
                        iterations)
  --gate-proj-bias-l2-loss-stage1-frac GATE_PROJ_BIAS_L2_LOSS_STAGE1_FRAC
                        fraction of the MoE (2D) gate_proj_bias L2 base weight
                        to reach at the end of stage 1 (1 = no stage-1
                        annealing)
  --gate-proj-bias-l2-loss-final-frac GATE_PROJ_BIAS_L2_LOSS_FINAL_FRAC
                        fraction of the MoE (2D) gate_proj_bias L2 base weight
                        to reach at the end of training during stage 2 (can
                        be above --gate-proj-bias-l2-loss-stage1-frac to
                        re-increase in stage 2)
  --bilinear-mlp-moe [BILINEAR_MLP_MOE]
                        disable the SiLU gate in Qwen3-style MoE MLPs only,
                        using raw bilinear gating in expert layers
  --router-z-loss-weight ROUTER_Z_LOSS_WEIGHT
                        weight for router z loss
  --router-z-loss-input-grad-scale ROUTER_Z_LOSS_INPUT_GRAD_SCALE
                        scaling factor for gradients to router input when
                        computing router z loss. Setting this to a value < 1.0
                        can help stabilize training by preventing large z-loss
                        gradients from destabilizing the router input
                        representations.
  --z-loss-demean-logits [Z_LOSS_DEMEAN_LOGITS]
                        use logits-demeaned router z loss
  --z-loss-penalize-mean-logits [Z_LOSS_PENALIZE_MEAN_LOGITS]
                        penalize mean logits in router z loss
  --aspect-ratio ASPECT_RATIO
                        model_dim = depth * aspect_ratio
  --head-dim HEAD_DIM   target head dimension for attention
  --max-seq-len MAX_SEQ_LEN
                        max context length
  --window-pattern WINDOW_PATTERN
                        sliding window pattern tiled across layers: L=full,
                        S=half context (e.g. 'SSL')
  --num-iterations NUM_ITERATIONS
                        explicit number of optimization steps (-1 = disable)
  --target-flops TARGET_FLOPS
                        calculate num_iterations to reach target_flops (-1 =
                        disable)
  --target-param-data-ratio TARGET_PARAM_DATA_RATIO
                        calculate num_iterations to maintain data:param ratio
                        (Chinchilla=20, -1 = disable)
  --use-moe-adjusted-scaling-params [USE_MOE_ADJUSTED_SCALING_PARAMS]
                        use MoE-adjusted scaling params instead of raw scaling
                        params when --target-param-data-ratio determines
                        target tokens
  --compile [COMPILE]   use torch.compile to speed up training (may cause
                        instability, use with caution)
  --device-batch-size DEVICE_BATCH_SIZE
                        per-device batch size. good number to reduce to
                        16,8,4,... if you OOM on VRAM.
  --total-batch-size TOTAL_BATCH_SIZE
                        total batch size in tokens. Must currently be
                        divisible by --device-batch-size * --max-seq-len * DDP
                        world size because each micro-step uses a fixed-shape
                        batch and padded rows would still affect auxiliary MoE
                        losses. Decent numbers are e.g. 524288. (-1 = auto-
                        compute optimal)
  --max-auto-grad-accum-steps MAX_AUTO_GRAD_ACCUM_STEPS
                        cap gradient accumulation steps when --total-batch-
                        size=-1 (-1 = disable cap)
  --embedding-lr EMBEDDING_LR
                        learning rate for embedding parameters (Adam)
  --unembedding-lr UNEMBEDDING_LR
                        learning rate for unembedding parameters (Adam)
  --weight-decay WEIGHT_DECAY
                        cautious weight decay for Transformer layer weights in
                        the Muon optimizer
  --matrix-lr MATRIX_LR
                        learning rate for matrix parameters (Muon)
  --matrix-optimizer {muon,aurora}
                        matrix optimizer for 2D parameters
  --muon-match-rms-adamw [MUON_MATCH_RMS_ADAMW]
                        use Kimi Muon LR scaling: 0.2*sqrt(max(out,in))
  --scalar-lr SCALAR_LR
                        learning rate for scalars (resid_lambdas, x0_lambdas)
  --adam-beta1 ADAM_BETA1
                        Adam beta1 for embedding/unembedding
  --adam-beta2 ADAM_BETA2
                        Adam beta2 for embedding/unembedding
  --lr-scheduler-skip-iters LR_SCHEDULER_SKIP_ITERS
                        number of initial iterations to skip for LR scheduling
                        (to allow for redoing warmup when resuming from a
                        later point in training)
  --lr-base-scale LR_BASE_SCALE
                        base scale for learning rate
  --warmup-ratio WARMUP_RATIO
                        ratio of iterations for LR warmup
  --warmdown-ratio WARMDOWN_RATIO
                        ratio of iterations for LR warmdown
  --final-lr-frac FINAL_LR_FRAC
                        final LR as fraction of initial LR
  --resume-from-step RESUME_FROM_STEP
                        resume training from this step (-1 = disable)
  --eval-every EVAL_EVERY
                        evaluate val bpb every N steps (-1 = disable)
  --eval-tokens EVAL_TOKENS
                        number of tokens to evaluate val loss on
  --core-metric-every CORE_METRIC_EVERY
                        evaluate CORE metric every N steps (-1 = disable)
  --core-metric-max-per-task CORE_METRIC_MAX_PER_TASK
                        examples per task for CORE metric
  --sample-every SAMPLE_EVERY
                        sample from model every N steps (-1 = disable)
  --save-every SAVE_EVERY
                        save checkpoints every N steps (-1 = only at end)
  --save-optimizer-state [SAVE_OPTIMIZER_STATE]
                        save optimizer shards alongside model checkpoints
  --delete-old-ckpts [DELETE_OLD_CKPTS]
                        after saving a checkpoint, delete all older
                        checkpoints based on step number
  --delete-old-ckpts-before-save
                        delete old checkpoints before saving the new
                        checkpoint; keeps file-size validation by snapshotting
                        the previous checkpoint sizes first
  --continue-to-chat-sft
                        after a successful base training run, exec
                        scripts.chat_sft from the final base checkpoint; when
                        launched under torchrun, each existing worker
                        continues in place with the same world size
  --continue-to-chat-sft-args CONTINUE_TO_CHAT_SFT_ARGS
                        extra CLI args forwarded to scripts.chat_sft when
                        --continue-to-chat-sft is set
  --model-tag MODEL_TAG
                        override model tag for checkpoint directory name
  --wandb-api-key-file WANDB_API_KEY_FILE
                        Weights & Biases API key file (optional). If provided,
                        sets WANDB_API_KEY for this run
  --log-grad-stats      log gradient statistics for MoE layers
  --log-interval LOG_INTERVAL
                        interval (in steps) for logging grad stats
  --debug [DEBUG]
